106void D3widget::initGeometry(){

117
118 // Atrybut pozycji: lokalizacja 0, 3 składowe, stride = 6 * sizeof(GLfloat)
119 glEnableVertexAttribArray(0);
120 glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(GLfloat), (void*)0);
121
122 // Atrybut koloru: lokalizacja 1, 3 składowe, offset = 3 * sizeof(GLfloat)
123 glEnableVertexAttribArray(1);
124 glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 6 * sizeof(GLfloat), (void*)(3 * sizeof(GLfloat)));
125
126 glBindVertexArray(0);
127}

135void D3widget::paintGL(){
136 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
137
138 QMatrix4x4 view;
139 view.translate(0,0,-5);
140 view.rotate(m_pitch * 180.0f/M_PI, -1,0,0);
141 view.rotate(m_roll * 180.0f/M_PI, 0,0,1);
142
143 QMatrix4x4 mvp = m_projection * view;
144 m_program.bind();
145 m_program.setUniformValue("u_mvp", mvp);
146
147 glBindVertexArray(m_vao);
148 glDrawElements(GL_TRIANGLES, sizeof(indices)/sizeof(GLuint),
149 GL_UNSIGNED_INT, nullptr);
150 glBindVertexArray(0);
151}
