38QByteArray SerialPortHandler::computeCRC(const QByteArray& dataLine)
39{
40 uint8_t crc = 0x00;
41 const uint8_t poly = 0x07;
42
43 for (auto b : dataLine) {
44 crc ^= static_cast<uint8_t>(b);
45 for (int bit = 0; bit < 8; ++bit) {
46 if (crc & 0x80) {
47 crc = static_cast<uint8_t>((crc << 1) ^ poly);
48 } else {
49 crc <<= 1;
50 }
51 }
52 }
53
54 QByteArray hex = QByteArray::number(crc, 16).toUpper();
55 if (hex.size() == 1)
56 hex.prepend('0');
57 return hex;
58}

125void SerialPortHandler::onReadyRead()
126{
127 if (!m_reading) return;
128
129 m_buffer.append(m_port.readAll());
130 int idx;
131 while ((idx = m_buffer.indexOf('\r')) != -1) {
132 QByteArray frame = m_buffer.left(idx).trimmed();
133 m_buffer.remove(0, idx + 1);
134
135 auto parts = frame.split(' ');
136 if (parts.size() == 5 && parts[0] == "A") {
137 QByteArray dataLine = parts.mid(0,4).join(' ') + ' ';
138 QByteArray recvCRC = parts[4];
139 QByteArray calcCRC = computeCRC(dataLine);
140
141 if (recvCRC != calcCRC) {
142 qCritical() << "CRC mismatch! recv=" << recvCRC
143 << " calc=" << calcCRC
144 << " frame=" << frame;
145 continue;
146 }
147
148 bool ok1, ok2, ok3;
149 int rawX = parts[1].toInt(&ok1);
150 int rawY = parts[2].toInt(&ok2);
151 int rawZ = parts[3].toInt(&ok3);
152
153 if (ok1 && ok2 && ok3) {
154 const double scale = 15893.0;
155 double x_g = rawX / scale;
156 double y_g = rawY / scale;
157 double z_g = rawZ / scale;
158
159 if (m_firstMeasurement) {
160 m_lastX = x_g;
161 m_lastY = y_g;
162 m_lastZ = z_g;
163 m_firstMeasurement = false;
164 } else {
165 if (fabs(x_g - m_lastX) < 0.01)
166 x_g = m_lastX;
167 if (fabs(y_g - m_lastY) < 0.01)
168 y_g = m_lastY;
169 if (fabs(z_g - m_lastZ) < 0.01)
170 z_g = m_lastZ;
171 }
172
173 m_lastX = x_g;
174 m_lastY = y_g;
175 m_lastZ = z_g;
176 emit newData(x_g, y_g, z_g);
177 } else {
178 qCritical() << "Parse error in frame (raw ints):" << frame;
179 }
180 } else {
181 qCritical() << "Bad frame format:" << frame;
182 }
183 }
184}
